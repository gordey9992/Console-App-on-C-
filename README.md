```cpp
#include <iostream>
using namespace std;

int main() {
    cout << "bool " << sizeof(bool) << endl;
    cout << "char " << sizeof(char) << endl;
    cout << "int " << sizeof(int) << endl;
    cout << "float " << sizeof(float) << endl;
    cout << "double " << sizeof(double) << endl;
    int age = 10;
    cout << age - 5 << ", " << age << " - 5 = 5" << endl;
    int x = 1;
    x = x + 5;
    cout << x / 2 << ", " << x << " : 2 = " << x / 2 << endl;
}
   ```
